    // APPROACH 2: Tabulation (Bottom-Up DP) - OPTIMAL
    // Time: O(n), Space: O(n) for DP array
    // REVISION NOTES:
    // - Base cases: dp[0]=1 (one way: stay), dp[1]=1 (one way: step 1)
    // - Recurrence: dp[i] = dp[i-1] + dp[i-2] (reach step i from i-1 or i-2)
    // - Key insight: This is a Fibonacci-like pattern
    int climbStairs(int n) {
        if (n == 0 || n == 1) {
            return 1;
        }
        
        vector<int> dp(n + 1);
        dp[0] = 1;  // 1 way to stay at ground
        dp[1] = 1;  // 1 way to reach step 1
        
        // For each step, we can reach it from previous step or two steps back
        for (int i = 2; i <= n; i++) {
            dp[i] = dp[i-1] + dp[i-2];
        }
        
        return dp[n];  // Return total ways to reach step n
    }
};
