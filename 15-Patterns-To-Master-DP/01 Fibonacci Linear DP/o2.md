// APPROACH 3: Tabulation (Bottom-Up DP) - OPTIMAL
// Time: O(n), Space: O(n)
// REVISION NOTES:
// - dp[i] = minimum cost to reach step i
// - From each step, we can go 1 or 2 steps ahead
// - Work backwards: dp[i] = cost[i] + min(dp[i+1], dp[i+2])
// - Start from bottom: minimum of reaching via step 0 or step 1
class Solution {
public:
    int minCostClimbingStairs(vector<int>& cost) {
        int n = cost.size();
        vector<int> dp(n + 1, 0);
        
        // dp[i] = min cost to reach step i
        // Fill from right to left (backward)
        for(int i = n - 1; i >= 0; i--) {
            dp[i] = cost[i] + min(dp[i+1], dp[i+2]);
        }
        
        // Can start from step 0 or step 1
        return min(dp[0], dp[1]);
    }
};
